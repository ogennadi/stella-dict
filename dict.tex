\section*{Suffixes}
  \display{-deen\+}{-deenu}
  \head{-d\h e\m en\m\+}
  \definition{Still doing the verb. Synonymous with \emph{ka na-(verb)}.}
  \example{I jinadeen\+?}{Are you still sleeping?}

  \display{-g\+de}{-gude}
  \head{-g\h\+d\h e}
  \definition{to continue doing the verb}
  \example{Rig\+de ihe k\d e I t\+latag\+d\d e}

  \display     {-yar\+}{-yaru}
  \head        {-y\l ar\h\+}
  \definition  {to do the verb too much; to over-(verb).}
  \example     {H\d e na-an\+yar\+ ya nri}{They give him too much food lit. they over-give him food.}

  \display{-zhie}{-zhie}
  \head{-zh\h i\h e}
  \definition{Do more of the verb.}
  \example{gotezhie}{buy more.}
