\section*{Suffixes}
  \display{-deen\+}{-deenu}
  \head{-d\h e\m en\m\+}
  \definition{Still doing the verb. Synonymous with \emph{ka na-(verb)}.}
  \example{I jinadeen\+?}{Are you still sleeping?}

  \display{-zhie}{-zhie}
  \head{-zh\h i\h e}
  \definition{Do more of the verb.}
  \example{gotezhie}{buy more.}
